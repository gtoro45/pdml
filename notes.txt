when a new pcap file is detected ==> start parallel threads 

thread a || b || c ==> {
    call zeek command to extract pcap logs (fork + exec)
    run formatting code for final csv
    store csv file path in queue (to be piped into a different process)
}

thread b || c || a ==> {
    call zeek command to extract pcap logs (fork + exec)
    run formatting code for final csv
    store csv file path in queue (to be piped into a different process)
}

thread c || a || b ==> {
    call zeek command to extract pcap logs (fork + exec)
    run formatting code for final csv
    store csv file path in queue (to be piped into a different process)
}
